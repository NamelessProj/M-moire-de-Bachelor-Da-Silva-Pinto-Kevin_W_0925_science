\newglossaryentry{opensource}{
    name={open source},
    description={Modèle de développement dans lequel le code source est public et peut être consulté, modifié et redistribué par tous, selon les conditions de sa licence.}
}

\newglossaryentry{python}{
    name={Python},
    description={Langage de programmation polyvalent, réputé pour sa syntaxe simple et lisible, utilisé notamment pour le développement web, l’automatisation, l’analyse de données et l’intelligence artificielle.}
}

\newglossaryentry{javascript}{
    name={JavaScript},
    description={Langage de programmation principalement utilisé pour rendre les pages web interactives et dynamiques.}
}

\newglossaryentry{frontend}{
    name={front-end},
    description={Partie visible d’une application ou d’un site web avec laquelle l’utilisateur interagit directement.}
}

\newglossaryentry{backend}{
    name={back-end},
    description={Partie d’une application qui gère la logique, les données et les traitements en arrière-plan.}
}

\newglossaryentry{intergiciel}{
    name={intergiciel},
    description={Logiciel intermédiaire qui permet à plusieurs applications, services ou systèmes de communiquer et de fonctionner ensemble.}
}
